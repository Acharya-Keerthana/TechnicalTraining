#include <stdio.h>
#include <math.h>

int main() {
    // Write C code here
    int s,i,cnt=0,x,r,sum=0,count=0,e;
    scanf("%d%d",&s,&e);
while(s<=e){
    cnt=0;
    for(i=2;i<=sqrt(s);i++){
        if(s%i==0){
            cnt++;
        }
    }
    if(cnt==0){
        x=s;
        
        while(x>0){
            r=x%10;
            if(r==3 || r==2 || r==5 || r==7){
                sum=sum+r;
            }
            x=x/10;
        }
        count=0;
    for(i=2;i<=sqrt(sum);i++){
        if(sum%i==0){
            count++;
        }
    }
    if(count==0){
        printf("%d ",s);
    }
    }
    s++;
}
    return 0;
}
