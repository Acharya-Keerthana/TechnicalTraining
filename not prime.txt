#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <limits.h>
#include <stdbool.h>

int main() {
    // Write C code here
    int n,i,j,s=INT_MAX,l=INT_MIN,k,count,flag=0,isnotprime=0,c;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        count=0;//remeber this
    for(k=2;k<=sqrt(a[i]);k++)
        {
            if((a[i]%k)==0){
                count++;
            }
        }
        if(count==0){
            flag=1;
            if(a[i]<s){
                s=a[i];
                
            }
            if(a[i]>l){
                l=a[i];
            }
        }
    }
    if(flag==0 || s==l){
        printf("\n prime number");
    }
        while(s<=l){
            count=0;
            for(j=2;j<=sqrt(s);j++){
                if(s%j==0){
                    count++;
                }
            }
            if(count==0){
                c=0;
                for(i=0;i<n;i++){
                if(s==a[i]){
                   c++;
                }
                }
                if(c==0){
                    isnotprime=1;
                    break;
                }
            }
            s++;
        }
        if(isnotprime=0){
            printf("\n Prime Array");
        }
        else{
            printf("\n Not the Prime Array");
        }
        
    return 0;
}
